#include<stdio.h>
int main()
{
	int n,i,j,sum=0;
	printf("\n Enter upto which number do u want to find perfect numbers:");
	scanf("%d",&n);
	for(i=1;i<=n;i++)
	{
		sum=0;
		for(j=1;j<i;j++)
		{
			if(i%j==0)
			{
				sum=sum+j;
			}
		}
		if(i==sum)
		{
			printf("%d\t",sum);
		}
		
	}
}
