#include<stdio.h>
int main()
{
	int n, counter=0;
	printf("\n Enter a number:");
	scanf("%d",&n);
	while(n!=0)
	{
		n=n/10;
		counter=counter+1;
	}
	printf("count of digits is :%d",counter);
}
